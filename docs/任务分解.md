```typescript
// Dashboard.tsx - handleAddChart 修复
const handleAddChart = useCallback((type: ChartConfig['type']) => {
  if (!currentDashboard) return;
  
  // 计算新图表位置：找到当前最大 y 值
  const maxY = currentDashboard.charts.reduce((max, chart) => {
    const chartBottom = (chart.gridLayout?.y || 0) + (chart.gridLayout?.h || 4);
    return Math.max(max, chartBottom);
  }, 0);
  
  const newChart: ChartConfig = {
    id: `chart-${Date.now()}`,
    type,
    title: `新图表-${type}`,
    datasetId: '',
    gridLayout: { x: 0, y: maxY, w: 6, h: 4 },
  };
  
  addChart(newChart);
  setAddingChart(false);
  chartForm.resetFields();
}, [currentDashboard, addChart, chartForm]);
```

**验收标准**：
- [ ] 不存在的仪表盘 ID 不再显示空白页，3秒内自动跳转回列表并提示
- [ ] API 错误时显示友好错误提示
- [ ] 添加图表后布局不再卡死
- [ ] 新建仪表盘后跳转流畅

### 任务 5：添加图表流程简化（优先级：P1）

**问题**：当前添加图表需要填写 datasetId、xAxis、yAxis 等多个字段，流程繁琐

**优化方案**：简化为两步：选择类型 → 输入标题 → 完成

**涉及文件**：
| 文件 | 修改内容 |
|------|---------|
| `src/pages/report/components/ChartTypeSelector.tsx` | 新建图表类型选择组件（卡片式选择器） |
| `src/pages/report/Dashboard.tsx` | 简化 Modal 弹窗，只保留标题输入 |

**具体修改**：

```tsx
// ChartTypeSelector.tsx
export const ChartTypeSelector: React.FC<{
  onSelect: (type: ChartConfig['type']) => void;
}> = ({ onSelect }) => {
  const chartTypes = [
    { type: 'line', label: '折线图', icon: <LineChartOutlined />, desc: '展示趋势变化' },
    { type: 'bar', label: '柱状图', icon: <BarChartOutlined />, desc: '对比分类数据' },
    { type: 'pie', label: '饼图', icon: <PieChartOutlined />, desc: '展示占比分布' },
    { type: 'table', label: '数据表', icon: <TableOutlined />, desc: '展示明细数据' },
    { type: 'kpi', label: 'KPI指标', icon: <DashboardOutlined />, desc: '展示核心指标' },
  ];

  return (
    <div style={{ display: 'grid', gridTemplateColumns: 'repeat(3, 1fr)', gap: 16 }}>
      {chartTypes.map(item => (
        <Card hoverable onClick={() => onSelect(item.type)} className="chart-type-card">
          <div style={{ fontSize: 36, marginBottom: 8 }}>{item.icon}</div>
          <div style={{ fontWeight: 500, color: '#fff' }}>{item.label}</div>
          <div style={{ color: '#a0a0a0', fontSize: 12 }}>{item.desc}</div>
        </Card>
      ))}
    </div>
  );
};
```

**验收标准**：
- [ ] 点击"添加图表"后显示卡片式类型选择器
- [ ] 选择类型后只弹出标题输入框
- [ ] 输入标题后图表自动创建（默认位置）
- [ ] 不再需要手动输入 datasetId、xAxis、yAxis

**预计工时**：2小时
